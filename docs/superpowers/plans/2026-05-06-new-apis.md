```python
from typing import Optional
import requests

_URL = "https://aa.usno.navy.mil/api/rstt/oneday"


def fetch_astronomical_twilight(lat: float, lon: float, date: str) -> Optional[str]:
    """天文薄明終了時刻（JST）を返す。取得失敗時は None。"""
    try:
        resp = requests.get(
            _URL,
            params={"date": date, "coords": f"{lat},{lon}", "tz": 9},
            timeout=10,
        )
        resp.raise_for_status()
        phenomena = resp.json()["properties"]["data"]["phenomena"]
        for p in phenomena:
            if p["phen"] == "EA":
                return p["time"]
        return None
    except Exception:
        return None
```

```python
from typing import Optional
import requests

_URL = "https://api.nasa.gov/planetary/apod"


def fetch_apod(api_key: str) -> Optional[tuple[str, str]]:
    """APOD の (image_url, title) を返す。動画の日・失敗時は None。"""
    try:
        resp = requests.get(_URL, params={"api_key": api_key}, timeout=10)
        resp.raise_for_status()
        data = resp.json()
        if data.get("media_type") != "image":
            return None
        return data["url"], data["title"]
    except Exception:
        return None
```

```python
from typing import Optional
import requests

_URL = "https://www.jma.go.jp/bosai/forecast/data/forecast/{area_code}.json"

_CODE_PENALTY = {1: 0, 2: -1, 3: -2, 4: -2}


def _code_to_penalty(code_str: str) -> int:
    try:
        bucket = int(code_str) // 100
        return _CODE_PENALTY.get(bucket, 0)
    except (ValueError, TypeError):
        return 0


def fetch_jma_night_penalty(area_code: str, date_jst: str) -> Optional[int]:
    """夜間（18時以降）の天気コードをスコア補正値として返す。失敗時は None。"""
    try:
        resp = requests.get(
            _URL.format(area_code=area_code), timeout=10
        )
        resp.raise_for_status()
        data = resp.json()
        time_series = data[0]["timeSeries"][0]
        time_defines = time_series["timeDefines"]
        codes = time_series["areas"][0]["weatherCodes"]

        # 18:00 JST 以降の最初のエントリを夜間天気として採用
        night_code = None
        for t, c in zip(time_defines, codes):
            if f"{date_jst}T18" in t or f"{date_jst}T21" in t:
                night_code = c
                break

        # 18時エントリがなければ最後のエントリを使う
        if night_code is None:
            night_code = codes[-1]

        return _code_to_penalty(night_code)
    except Exception:
        return None
```

```python
    cloud_cover: int, visibility: int, seeing: int, transparency: int,
    weather_penalty: int = 0,
) -> int:
    score = 5
    if cloud_cover > 80:
        score -= 2
    elif cloud_cover > 50:
        score -= 1
    if visibility < 10000:
        score -= 1
    if seeing >= 5:
        score -= 1
    if transparency >= 6:
        score -= 1
    score += weather_penalty
    return max(1, score)
```

```python
def main() -> None:
    cfg = config.load()
    now_utc = datetime.now(timezone.utc)
    date_jst = datetime.now(JST).date().isoformat()
    sky_error = None

    try:
        conditions = fetch_sky_conditions(cfg.LOCATION_LAT, cfg.LOCATION_LON)
    except Exception as e:
        sky_error = e

    if sky_error is not None:
        send_messages(cfg.LINE_CHANNEL_ACCESS_TOKEN, cfg.LINE_NOTIFY_TARGETS, f"⚠️ 気象データ取得失敗: {sky_error}")
        return

    moon_age, moonrise, planets, meteor_showers = get_astro_data(
        cfg.LOCATION_LAT, cfg.LOCATION_LON, now_utc
    )

    try:
        astro_data = fetch_7timer_astro(cfg.LOCATION_LAT, cfg.LOCATION_LON)
    except Exception:
        astro_data = {}

    try:
        constellations = fetch_constellations(cfg.LOCATION_LAT, cfg.LOCATION_LON, date_jst, cfg.HOSHIMIRU_API_TOKEN)
    except Exception:
        constellations = []

    twilight_end = fetch_astronomical_twilight(cfg.LOCATION_LAT, cfg.LOCATION_LON, date_jst)
    weather_penalty = fetch_jma_night_penalty(cfg.JMA_AREA_CODE, date_jst) or 0
    apod = fetch_apod(cfg.NASA_APOD_API_KEY)

    message = format_message(

        meteor_showers, constellations, twilight_end, weather_penalty,
    )
    send_messages(cfg.LINE_CHANNEL_ACCESS_TOKEN, cfg.LINE_NOTIFY_TARGETS, message)

    if apod:

        send_image_message(cfg.LINE_CHANNEL_ACCESS_TOKEN, cfg.LINE_NOTIFY_TARGETS, image_url)
```
